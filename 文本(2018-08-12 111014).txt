import sysm = input()a = []for i in range(int(m)):    line = sys.stdin.readline().strip()    a.append(line)begin = []end = []new = []for item in a:    b = item.replace(',',' ')    b = b.replace(';',' ')    new.append(b.split())smallist = []for item in new:    for i in range(len(item)):        c = []        if i%2 == 0:            begi = int(item[i])            endi = int(item[i+1])            c.append(begi)            c.append(endi)            smallist.append(c)for i in range(len(smallist)):    begin.append(smallist[i][0])    end.append(smallist[i][1])for i in range(len(begin)):    for j in range(i,len(end)):        if begin[i]> begin[j]:            temp = begin[i]            temp2 = end[i]            begin[i] = begin[j]            end[i] = end[j]            begin[j] = temp            end[j] = temp2            for i in range(len(begin)):    for j in range(i,len(end)):        if begin[j]<= end[i]:            end[i] = end[j]count = []shuchu = []for i in range(len(begin)):    if end[i] not in count:        count.append(end[i])        shuchu.append(str(begin[i])+','+ str(end[i]))aa= ''for i in range(len(shuchu)):    if i < len(shuchu)-1:        aa += shuchu[i]+';'    else:        aa +=shuchu[i]print(aa)
